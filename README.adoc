= Test Web API

[source,yml]
.API定義
----
schemes:
- "http"
paths:
  /test:
    get:
      tags:
      - "test"
      summary: "Output static strings"
      description: "test APvimI"
      responses:
        200:
          description: "Hello, test API !"
  /uinfo:
    get:
      tags:
      - "uinfo"
      summary: "Get user infomation"
      description: "Output user id and content"
      parameters:
        -
          name: "name"
          in: "query"
          required: false
          type: "string"
          description: ""
      responses:
        200:
          description: "successful operation"
          schema:
            type: "object"
            properties:
              id:
                type: "integer"
                format: "int64"
              content:
                type: "string"
                example: "Hello, API {name}!"
----
